8   import org.springframework.boot.actuate.health.HealthIndicator;
9   import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
10  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
11  import org.springframework.cloud.config.client.ConfigServerHealthIndicator;
12  import org.springframework.cloud.config.client.ConfigServicePropertySourceLocator;
13  import org.springframework.core.env.Environment;
14  import org.springframework.stereotype.Component;
15  import org.springframework.util.ReflectionUtils;
16  
17  import java.lang.reflect.Field;
18  
19  /**
20   * Created by Jin Yuliang on 16/11/17.
21   * This component is a fix to issue: https://github.com/spring-cloud/spring-cloud-config/issues/182
22   */
23  @Component
24  @ConditionalOnClass({HealthIndicator.class, ConfigServicePropertySourceLocator.class})
25  @ConditionalOnProperty(value = "health.config.enabled", matchIfMissing = true)
26  @Slf4j
27  public class ConfigServerPostProcessor implements BeanPostProcessor {
28  
29    @Autowired
30    private Environment environment;
31  
32    @Override
33    public Object postProcessBeforeInitialization(final Object bean, final String beanName) throws BeansException {
34      return bean;
35    }
36  
37    @Override
38    public Object postProcessAfterInitialization(final Object bean, final String beanName) throws BeansException {
39      if (ConfigServerHealthIndicator.class.isAssignableFrom(bean.getClass())) {
40        log.info("Config Server Bean: {} was processed.", beanName);
41  
42        final Boolean env = setField(bean, ConfigServerHealthIndicator.class, "env", this.environment);
43        final Boolean environment = setField(bean, ConfigServerHealthIndicator.class, "environment", this.environment);
44        if (!env && !environment) {
45          log.warn("env and environment not found in ConfigServerHealthIndicator.class");
46        }
47      }
48      return bean;
49    }
50  
51    static Boolean setField(final Object object, final Class<?> objectType, final String field, final Object value) {
52      final Field fieldFound = ReflectionUtils.findField(objectType, field);
53      final Boolean result;
54      if (fieldFound != null) {
55        fieldFound.setAccessible(true);
56        ReflectionUtils.setField(fieldFound, object, value);
57        result = Boolean.TRUE;
58      } else {
59        result = Boolean.FALSE;
60      }
61      return result;
62    }
63  }
